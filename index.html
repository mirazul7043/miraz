// ========================= // সম্পূর্ণ বাংলা ইনভেন্টরি সিস্টেম (Node.js + MongoDB) // ========================= // ---------- BACKEND (server.js) ---------- const express = require('express'); const mongoose = require('mongoose'); const cors = require('cors'); const app = express(); app.use(express.json()); app.use(cors()); mongoose.connect('mongodb://127.0.0.1:27017/inventoryDB') .then(() => console.log('ডাটাবেজ সংযুক্ত হয়েছে')) .catch(err => console.log(err)); // ইউজার স্কিমা const userSchema = new mongoose.Schema({ username: String, password: String }); const User = mongoose.model('User', userSchema); // পণ্য স্কিমা const itemSchema = new mongoose.Schema({ nam: String, poriman: Number, dam: Number }); const Item = mongoose.model('Item', itemSchema); // -------- রেজিস্টার -------- app.post('/api/register', async (req, res) => { const user = new User(req.body); await user.save(); res.json({ message: 'রেজিস্ট্রেশন সম্পন্ন হয়েছে' }); }); // -------- লগইন -------- app.post('/api/login', async (req, res) => { const { username, password } = req.body; const user = await User.findOne({ username, password }); if (user) { res.json({ success: true, message: 'লগইন সফল' }); } else { res.json({ success: false, message: 'ভুল তথ্য' }); } }); // -------- পণ্য যোগ -------- app.post('/api/items', async (req, res) => { const item = new Item(req.body); await item.save(); res.json(item); }); // -------- সব পণ্য -------- app.get('/api/items', async (req, res) => { const items = await Item.find(); res.json(items); }); // -------- পণ্য মুছুন -------- app.delete('/api/items/:id', async (req, res) => { await Item.findByIdAndDelete(req.params.id); res.json({ message: 'মুছে ফেলা হয়েছে' }); }); app.listen(5000, () => console.log('সার্ভার চালু আছে (port 5000)')); // ---------- LOGIN PAGE (login.html) ---------- /* লগইন
